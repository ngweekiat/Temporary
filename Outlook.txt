Sub OpenMsgTemplate()
    Dim OutApp As Object
    Dim olItem As Object
    Dim msgPath As String
    
    ' Build relative path: Excel's folder + \Automation\YourTemplate.msg
    msgPath = ThisWorkbook.Path & "\Automation\YourTemplate.msg"
    
    ' Launch Outlook if not already running
    On Error Resume Next
    Set OutApp = GetObject(, "Outlook.Application")
    If OutApp Is Nothing Then
        Set OutApp = CreateObject("Outlook.Application")
    End If
    On Error GoTo 0
    
    ' Open the .msg file
    Set olItem = OutApp.Session.OpenSharedItem(msgPath)
    
    ' Show it in Outlook
    olItem.Display
End Sub
