Sub OpenMsgTemplate()
    Dim OutApp As Object
    Dim msgPath As String
    
    ' Build relative path: ParentFolder/Automation/YourTemplate.msg
    msgPath = ThisWorkbook.Path & "\Automation\YourTemplate.msg"
    
    ' Create Outlook object
    Set OutApp = CreateObject("Outlook.Application")
    
    ' Open the .msg file (like double-clicking it in Explorer)
    OutApp.Session.OpenSharedItem msgPath
    
    ' Clean up
    Set OutApp = Nothing
End Sub
