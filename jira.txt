Sub GetHTML_XHR_SSO()
    Dim http As Object
    Dim URL As String
    
    URL = "https://yourcompanysite/page"
    
    Set http = CreateObject("MSXML2.XMLHTTP")
    http.Open "GET", URL, False
    http.setRequestHeader "Content-Type", "text/html"
    http.setRequestHeader "Cache-Control", "no-cache"
    
    ' Let it use current logged-in credentials
    http.setOption 2, 13056 ' ignore SSL errors if needed
    
    http.Send
    Debug.Print http.ResponseText
End Sub
