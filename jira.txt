Sub GetHTML_XHR_SSO()
    Dim http As Object
    Dim URL As String
    
    URL = "https://yourcompanysite/page"
    
    ' Use ServerXMLHTTP instead of XMLHTTP
    Set http = CreateObject("MSXML2.ServerXMLHTTP.6.0")
    
    ' This tells it to use your current Windows credentials (for NTLM/Kerberos SSO)
    http.setOption 3, "AUTOLOGIN_ONLY_SERVER"
    
    ' Open and send request
    http.Open "GET", URL, False
    http.Send
    
    Debug.Print http.ResponseText
End Sub
