'-----------------------------
' API function to copy text
'-----------------------------
Private Sub CopyTextToClipboard(sText As String)
    Dim hGlobal As LongPtr
    Dim lpGlobal As LongPtr
    
    OpenClipboard 0
    EmptyClipboard
    
    hGlobal = GlobalAlloc(GMEM_MOVEABLE, Len(sText) + 1)
    lpGlobal = GlobalLock(hGlobal)
    
    If lpGlobal <> 0 Then
        lstrcpy ByVal lpGlobal, ByVal sText
        GlobalUnlock hGlobal
        SetClipboardData CF_TEXT, hGlobal
    End If
    
    CloseClipboard
End Sub

Sub GetWebPageHTML()
    Dim Http As Object
    Dim HTMLDoc As Object
    Dim URL As String
    Dim HTMLContent As String
    
    ' URL of the webpage
    URL = "https://www.example.com"
    
    ' Create the XMLHTTP object
    Set Http = CreateObject("MSXML2.XMLHTTP")
    
    ' Open a GET request
    Http.Open "GET", URL, False
    Http.Send
    
    ' Get the response text (HTML source)
    HTMLContent = Http.ResponseText
    
    ' Print to Immediate Window (Ctrl+G to view in VBA editor)
    Debug.Print HTMLContent
    
    ' Optionally load into an HTML document for parsing
    Set HTMLDoc = CreateObject("HTMLFILE")
    HTMLDoc.body.innerHTML = HTMLContent
    
    ' Example: print the page title
    Debug.Print HTMLDoc.getElementsByTagName("title")(0).innerText
End Sub
