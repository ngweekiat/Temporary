Sub OpenNewTab_AndRefresh()
    Dim b As AutomateBrowser
    Set b = New_AutomateBrowser
    
    ' Launch Chrome (once)
    b.launch whichBrowser:=Chromium
    
    ' Open first tab and navigate
    Dim firstTab As cdpConnection
    Set firstTab = b.newTab
    firstTab.Page.navigate "https://www.google.com"
    firstTab.WaitForPageLoaded
    
    ' Refresh the first tab every 8 seconds in the background
    Dim startTime As Double
    startTime = Timer
    Do While Timer - startTime < 60 ' refresh for 1 minute
        firstTab.Page.reload
        Application.Wait Now + TimeValue("00:00:08")
    Loop
    
    ' Open a new tab without closing the session
    Dim newTab As cdpConnection
    Set newTab = b.newTab
    newTab.Page.navigate "https://www.bing.com"
    newTab.WaitForPageLoaded
    
    ' Optional: refresh the new tab as well
    newTab.Page.reload
    
    ' Keep browser open; do not call b.Quit
End Sub
