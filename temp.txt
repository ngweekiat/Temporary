Sub OpenNewTabAndNavigateGoogle()
    Dim b As AutomateBrowser
    Set b = New_AutomateBrowser

    ' Launch Chromium if not already running (optional)
    b.launch whichBrowser:=Chromium

    ' Create a new tab (target)
    Dim newTab As cdpTarget
    Set newTab = b.targets.create

    ' Attach to the new tab
    Dim session As cdpSession
    Set session = b.targets.attach(newTab)

    ' Navigate to Google
    session.send "Page.navigate", "{""url"":""https://www.google.com""}"

    ' Wait for page to load (optional)
    session.send "Page.enable"
End Sub
