Function SanitizeInputText(inputText As String) As String
    Dim txt As String
    txt = LCase(inputText)
    
    Dim words() As String
    words = Split(txt, " ")
    
    Dim wdApp As Object
    Set wdApp = CreateObject("Word.Application")
    wdApp.Visible = False
    
    Dim i As Long, w As String
    For i = LBound(words) To UBound(words)
        w = words(i)
        If w <> "" And w Like "*[A-Za-z]*" Then
            If Not wdApp.CheckSpelling(w) Then
                words(i) = "[Redacted]"
            End If
        End If
    Next i
    
    wdApp.Quit
    Set wdApp = Nothing
    
    txt = Join(words, " ")
    
    Dim regEx As Object
    Set regEx = CreateObject("VBScript.RegExp")
    
    With regEx
        .Pattern = "[^a-z\s'\-\[\]]+"  ' Allow letters, spaces, apostrophes, hyphens, and brackets
        .Global = True
    End With
    txt = regEx.Replace(txt, "[Redacted]")
    
    SanitizeInputText = txt
End Function
