Function SanitizeSensitiveText(inputText As String) As String
    Dim txt As String
    txt = inputText
    
    ' Remove all non-letter characters (keep spaces)
    txt = Replace(txt, vbTab, " ") ' replace tabs with space
    Dim regEx As Object
    Set regEx = CreateObject("VBScript.RegExp")
    regEx.Pattern = "[^A-Za-z\s]"
    regEx.Global = True
    txt = regEx.Replace(txt, "")
    
    ' Split text into words
    Dim words() As String
    words = Split(txt, " ")
    
    ' Create Word application object
    Dim wdApp As Object
    Set wdApp = CreateObject("Word.Application")
    wdApp.Visible = False
    
    Dim i As Long
    Dim w As String
    For i = LBound(words) To UBound(words)
        w = words(i)
        ' Only process non-empty words
        If w <> "" Then
            ' Redact words not in dictionary
            If Not wdApp.CheckSpelling(w) Then
                words(i) = "[Redacted]"
            End If
        End If
    Next i
    
    ' Close Word
    wdApp.Quit
    Set wdApp = Nothing
    
    ' Join words back into text
    txt = Join(words, " ")
    
    SanitizeSensitiveText = txt
End Function
