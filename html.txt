Function ExtractSpanTextOnly(ByVal html As String, ByVal cssSelector As String) As String
    Dim htmlDoc As Object
    Dim elem As Object
    Dim txt As String
    
    Set htmlDoc = CreateObject("HTMLFILE")
    htmlDoc.body.innerHTML = html
    
    Set elem = htmlDoc.querySelector(cssSelector)
    If Not elem Is Nothing Then
        txt = elem.innerText
        ' Remove anything in parentheses
        If InStr(txt, "(") > 0 Then
            txt = Left(txt, InStr(txt, "(") - 1)
        End If
        ExtractSpanTextOnly = Trim(txt)
    Else
        ExtractSpanTextOnly = vbNullString
    End If
End Function
