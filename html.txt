Option Explicit

' Late binding version (no reference needed)
Function ExtractSmallQuiet(ByVal html As String) As String
    Dim htmlDoc As Object
    Dim elem As Object
    
    ' Load into HTML parser
    Set htmlDoc = CreateObject("HTMLFILE")
    htmlDoc.body.innerHTML = html
    
    ' Grab the span with class="small quiet"
    Set elem = htmlDoc.querySelector("span.small.quiet")
    
    If Not elem Is Nothing Then
        ' InnerText â†’ just the human-readable text
        ExtractSmallQuiet = elem.innerText
    Else
        ExtractSmallQuiet = vbNullString
    End If
End Function
