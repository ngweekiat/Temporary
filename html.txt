Sub Edge_FullHTML()
    Dim b As AutomateBrowser
    Dim html As String

    Set b = New_AutomateBrowser
    b.launch whichBrowser:=browserType.Edge, url:="https://www.google.com/", killWithoutAsking:=True

    ' Wait for page scripts to run (adjust if needed)
    Application.Wait (Now + TimeValue("0:00:05"))

    ' Grab full DOM
    html = CStr(b.jsEval("document.documentElement.outerHTML"))
    Debug.Print html

    b.Quit
End Sub
